No matching records found.

"; return; } let totalOvertime = 0; results.forEach(record => { totalOvertime += parseInt(record.overtime); reportDiv.innerHTML += `

`; }); reportDiv.innerHTML += `

Total Overtime: ${totalOvertime} hours

`; }
